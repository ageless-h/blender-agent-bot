*class* bpy.types.FileSelectParams(*bpy\_struct*)[#](#bpy.types.FileSelectParams "Link to this definition")
:   File Select Parameters

    directory[#](#bpy.types.FileSelectParams.directory "Link to this definition")
    :   Directory displayed in the file browser

        Type:
        :   string, default “”, (never None)

    display\_size[#](#bpy.types.FileSelectParams.display_size "Link to this definition")
    :   Change the size of thumbnails

        Type:
        :   int in [16, 256], default 96

    display\_size\_discrete[#](#bpy.types.FileSelectParams.display_size_discrete "Link to this definition")
    :   Change the size of thumbnails in discrete steps

        Type:
        :   enum in [‘TINY’, ‘SMALL’, ‘NORMAL’, ‘BIG’, ‘LARGE’], default ‘TINY’

    display\_type[#](#bpy.types.FileSelectParams.display_type "Link to this definition")
    :   Display mode for the file list

        * `LIST_VERTICAL`
          Vertical List – Display files as a vertical list.
        * `LIST_HORIZONTAL`
          Horizontal List – Display files as a horizontal list.
        * `THUMBNAIL`
          Thumbnails – Display files as thumbnails.

        Type:
        :   enum in [‘LIST\_VERTICAL’, ‘LIST\_HORIZONTAL’, ‘THUMBNAIL’], default ‘LIST\_VERTICAL’

    filename[#](#bpy.types.FileSelectParams.filename "Link to this definition")
    :   Active file in the file browser

        Type:
        :   string, default “”, (never None)

    filter\_glob[#](#bpy.types.FileSelectParams.filter_glob "Link to this definition")
    :   UNIX shell-like filename patterns matching, supports wildcards (‘\*’) and list of patterns separated by ‘;’

        Type:
        :   string, default “”, (never None)

    filter\_id[#](#bpy.types.FileSelectParams.filter_id "Link to this definition")
    :   Which ID types to show/hide, when browsing a library

        Type:
        :   [`FileSelectIDFilter`](FileSelectIDFilter.md#bpy.types.FileSelectIDFilter "bpy.types.FileSelectIDFilter"), (readonly, never None)

    filter\_search[#](#bpy.types.FileSelectParams.filter_search "Link to this definition")
    :   Filter by name or tag, supports ‘\*’ wildcard

        Type:
        :   string, default “”, (never None)

    recursion\_level[#](#bpy.types.FileSelectParams.recursion_level "Link to this definition")
    :   Numbers of dirtree levels to show simultaneously

        * `NONE`
          None – Only list current directory’s content, with no recursion.
        * `BLEND`
          Blend File – List .blend files’ content.
        * `ALL_1`
          One Level – List all sub-directories’ content, one level of recursion.
        * `ALL_2`
          Two Levels – List all sub-directories’ content, two levels of recursion.
        * `ALL_3`
          Three Levels – List all sub-directories’ content, three levels of recursion.

        Type:
        :   enum in [‘NONE’, ‘BLEND’, ‘ALL\_1’, ‘ALL\_2’, ‘ALL\_3’], default ‘NONE’

    show\_details\_datetime[#](#bpy.types.FileSelectParams.show_details_datetime "Link to this definition")
    :   Show a column listing the date and time of modification for each file

        Type:
        :   boolean, default False

    show\_details\_size[#](#bpy.types.FileSelectParams.show_details_size "Link to this definition")
    :   Show a column listing the size of each file

        Type:
        :   boolean, default False

    show\_hidden[#](#bpy.types.FileSelectParams.show_hidden "Link to this definition")
    :   Show hidden dot files

        Type:
        :   boolean, default False

    sort\_method[#](#bpy.types.FileSelectParams.sort_method "Link to this definition")
    :   Type:
        :   enum in [Fileselect Params Sort Items](../../../appendix/bpy_types_enum_items/fileselect_params_sort_items.md#rna-enum-fileselect-params-sort-items), default ‘FILE\_SORT\_ALPHA’

    title[#](#bpy.types.FileSelectParams.title "Link to this definition")
    :   Title for the file browser

        Type:
        :   string, default “”, (readonly, never None)

    use\_filter[#](#bpy.types.FileSelectParams.use_filter "Link to this definition")
    :   Enable filtering of files

        Type:
        :   boolean, default False

    use\_filter\_asset\_only[#](#bpy.types.FileSelectParams.use_filter_asset_only "Link to this definition")
    :   Hide .blend files items that are not data-blocks with asset metadata

        Type:
        :   boolean, default False

    use\_filter\_backup[#](#bpy.types.FileSelectParams.use_filter_backup "Link to this definition")
    :   Show .blend1, .blend2, etc. files

        Type:
        :   boolean, default False

    use\_filter\_blender[#](#bpy.types.FileSelectParams.use_filter_blender "Link to this definition")
    :   Show .blend files

        Type:
        :   boolean, default False

    use\_filter\_blendid[#](#bpy.types.FileSelectParams.use_filter_blendid "Link to this definition")
    :   Show .blend files items (objects, materials, etc.)

        Type:
        :   boolean, default False

    use\_filter\_folder[#](#bpy.types.FileSelectParams.use_filter_folder "Link to this definition")
    :   Show folders

        Type:
        :   boolean, default False

    use\_filter\_font[#](#bpy.types.FileSelectParams.use_filter_font "Link to this definition")
    :   Show font files

        Type:
        :   boolean, default False

    use\_filter\_image[#](#bpy.types.FileSelectParams.use_filter_image "Link to this definition")
    :   Show image files

        Type:
        :   boolean, default False

    use\_filter\_movie[#](#bpy.types.FileSelectParams.use_filter_movie "Link to this definition")
    :   Show movie files

        Type:
        :   boolean, default False

    use\_filter\_script[#](#bpy.types.FileSelectParams.use_filter_script "Link to this definition")
    :   Show script files

        Type:
        :   boolean, default False

    use\_filter\_sound[#](#bpy.types.FileSelectParams.use_filter_sound "Link to this definition")
    :   Show sound files

        Type:
        :   boolean, default False

    use\_filter\_text[#](#bpy.types.FileSelectParams.use_filter_text "Link to this definition")
    :   Show text files

        Type:
        :   boolean, default False

    use\_filter\_volume[#](#bpy.types.FileSelectParams.use_filter_volume "Link to this definition")
    :   Show 3D volume files

        Type:
        :   boolean, default False

    use\_library\_browsing[#](#bpy.types.FileSelectParams.use_library_browsing "Link to this definition")
    :   Whether we may browse Blender files’ content or not

        Type:
        :   boolean, default False, (readonly)

    use\_sort\_invert[#](#bpy.types.FileSelectParams.use_sort_invert "Link to this definition")
    :   Sort items descending, from highest value to lowest

        Type:
        :   boolean, default False
